-- Fonction calculant deux résistances en parallèle
para(R1,R2) = R1*R2/(R1+R2)
 
-- Pont diviseur de résistances
pontdiv(E,R1,R2)=E*R2/(R1+R2)
 
-- Définition d'un vecteur à points équidistants
linspace(deb,fin,n)[i]=deb+(fin-deb)/(n-1)*(i-1) dim[n]
 
-- Interpolation
interplin(A,B,yo)=k:=1,nn:=count(B),
((xo:=A[k]+(A[k+1]-A[k])/(B[k+1]-B[k])*(yo-B[k]),k:=nn) when B[k+1]>yo,k:=k+1) while 
k<nn , xo
 
 
 
--****** Exercice 1 *******
 
-- 1.
-- Tension d'alim : E
E:=12:
 
-- Définition de la caractéristique d'entrée du Fet
-- Courant de drain maxi
Idss=8m
-- Tension de pincement (pour Id=0)
Vp=-4.5
 
-- Définition du vecteur Vgs de Vp (négatif) à zéro
Vgs=linspace(Vp,0,101)
 
-- Calcul du vecteur Id
Id=Idss*(1-Vgs/Vp)^2
 
-- Tracé Id=f(Vgs)
plotline {Vgs,Id}
 
-- Point Ido de polarisation
Ido=2m
 
-- Recherche du Vgso correspondant
Vgso=interplin(Vgs,Id,Ido)
Vgso:-2.25
 
 
-- 2.
-- Le potentiel Vgso est celui qu'on retrouve aux bornes de la résistance Rs parcourue 
par le courant de drain Ido. On a donc :
Rs=-Vgso/Ido
Rs/1K:1.12
 
 
-- 3.
-- Tension utile (Vds du fet + Vrd) : Vut (attention : Vgso<0)
Vut=E+Vgso
Vut:9.75
 
-- Tension aux bornes de Rd : Vrd
Vrd=Vut/2
Vrd:4.88
 
-- Calcul de Rd
Rd=Vrd/Ido
Rd/1K:2.44
 
 
-- 4.
-- Graphiquement, on trace la tangente au point de fonctionnement, et on mesure la 
pente de cette droite en prenant deux points de la tangente (pas de la courbe Id=f(Vgs) 
!)
-- On a aussi la formule de cours (g(Vgs) est la dérivée de la courbe Id=f(Vgs)) :
g=2*Idss/Vp*(Vgs/Vp-1)
 
-- Tracé de g(Vgs)
newaxis
plotline{Vgs,g}
 
-- Au point de polarisation Vgso :
go=2*Idss/Vp*(Vgso/Vp-1)
go/1m:1.78
 
 
 
 
 
--****** Exercice 2 *******
 
-- 1.
-- Tension d'alim : E
E:=15:
-- Courant de repos de collecteur : Ico
Ico=0.5m
-- Résistances de base : Rb1 et Rb2
Rb1=270K
Rb2=100K
-- Tension base émetteur au repos du transistor : Vbeo
Vbeo=0.7
-- Gain en courant du transistor : beta
beta=150
 
-- Red n'intervient pas dans la polarisation, car elle est en série avec un condensateur, 
d'impédance infinie en continu.
 
 
 
-- 2.
-- Courant de base Ibo
Ibo=Ico/beta
 
-- Tension base-masse (calcul tenant compte du générateur complet de thévenin du pont 
de base)
Vbmo=pontdiv(E,Rb1,Rb2)-para(Rb1,Rb2)*Ibo
Vbmo:3.81
 
-- Nota : pontdiv(E,Rb1,Rb2) est la tension à vide du pont diviseur Rb1, Rb2 (source 
de tension du générateur de thévenin équivalent) et para(Rb1,Rb2) son impédance interne 
(du géné de thévenin)
 
-- Tension émetteur-masse Vemo
Vemo=Vbmo-Vbeo
Vemo:3.11
 
-- Résistance Res :
Res=Vemo/Ico
Res/1K:6.22
 
 
-- 3.
-- Tension aux bornes de Rc : Vrc
Vrc=(E-Vemo)/2
Vrc:5.94
 
-- Valeur de Rc
Rc=Vrc/Ico
Rc/1K:11.89
 
 
-- 4.
-- Voir cours et exos
-- C'est le schéma équivalent du montage à charge répartie, avec la résistance d'émetteur 
Re remplacée par Red//Res
 
 
-- 5.
-- On pose Re=Res//Red : sur le schéma équivalent, ces deux résistances se trouvent 
en parallèle.
 
-- L'expression littérale du gain est :
-- Avo=-Rc/(h11/beta+Re)
-- Le gain est une donnée de l'énoncé
Avo:=-20:
 
 
-- 6.
-- Calcul de h11
h11=26m/Ibo
h11/1K:7.80
 
-- Calcul de Re d'après l'expression du gain (attention : Avo est négatif)
Re=-Rc/Avo-h11/beta
Re:542.46
 
-- Calcul de Red : Re=Red//Res, d'où :
Red=1/(1/Re-1/Res)
Red:594.27
 
 
-- 7.
Ze=para(para(Rb1,Rb2),(h11+beta*Re))
Ze/1K:40.13
-- Cette valeur est moyenne, mais meilleure que celle d'un émetteur commun (ça serait 
h11=7,8k )
 
-- 8.
-- Avec Res, on polarise le montage sans se soucier des petits signaux, et ensuite, 
pour l'alternatif, on vient placer une résistance Red en parallèle avec Res pour 
abaisser la valeur de Re. On notera que Red peut être un potentiomètre dont la valeur 
peut varier de 0 (on a alors un émetteur commun) à une valeur assez forte (montage à charge 
répartie). On peut ainsi faire facilement varier le gain en petits signaux sans se 
soucier de la polarisation, car on a mis un condensateur en série avec Res !
-- Ce réseau (Red, Ce) permet de découpler le fonctionnement en continu de celui en 
alternatif.
-- Un charge répartie qui aurait le même gain en tension serait polarisé avec un pont 
de base très dissymétrique (Rb1>>Rb2), car on aurait Res faible, donc une faible 
chute de tension à ses bornes. Ceci fera chuter l'impédance d'entrée du montage.
-- De plus, le gain en tension ne serait pas facilement réglable : il faudrait revoir 
toute la polarisation à chaque changement !
 
 
 
 
 
 
 
 
--****** Exercice 3 *******
 
-- 1.
-- Données
Zc:=5K:
Ze:=47K:
Zs:=22:
Zu:=100:
Av:=10:
-- Attention : Ve est donnée en crête à crête, alors qu'on demande la tension de sortie 
crête !!! Il y a un facteur 2 à ne pas oublier !
Ve:=1:
 
-- Il ne faut pas tenir compte de l'adaptation d'impédance en entrée, car la tension 
mesurée est celle en charge !
-- On tiendra uniquement compte de l'adaptation en sortie
Vs=pontdiv(Av*Ve/2,Zs,Zu)
Vs:4.10
 
 
 
-- 2.
-- Si l'ampli avait une impédance d'entrée infinie :
-- on aurait Ve=Vcapt*Ze/(Zc+Ze)
-- On en déduit Vcapt, qui est la tension à vide (donc sur charge infinie) recherchée
Vcapt=Ve*(Zc+Ze)/Ze
Vcapt:1.11
 
 
-- 3.
-- Dans ce cas, on aurait simplement (attention au facteur 2 de transformation crête 
à crête en crête) :
Vsth=Vcapt/2*Av
Vsth:5.53
 
 
-- 4.
-- La perte (en %) due à l'adaptation d'impédance vaut
perte=(Vsth-Vs)/Vsth*100
perte:25.91%
